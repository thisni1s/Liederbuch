\beginsong{Auf dem Donnerbalken}[wuw={Jürgen Fastje}, oedt={70}]

\beginverse
\[C]Auf dem Donnerbalken saßen zwei Gestalten,
und sie schrien nach \[G]Klopapier, Klopapier!
\endverse

\beginverse
\[G]Und dann kam der dritte, setzt sich in die Mitte,
und sie schrien nach \[C]Klopapier, Klopapier!
\endverse

\beginverse
\[C]Und dann kam der vierte, der sich gleich beschmierte,
und sie schrien nach \[G]Klopapier, Klopapier!
\endverse

\beginverse
\[G]Und dann kam der fünfte, der die Nase rümpfte,
und sie schrien nach \[C]Klopapier, Klopapier!
\endverse

\beginverse
\[C]Und dann kam der sechste, der sich gleich bekleckste,
und sie schrien nach \[G]Klopapier, Klopapier!
\endverse

\beginverse
\[G]Und dann kam der siebte, als der Balken wippte,
und sie schrien nach \[C]Klopapier, Klopapier!
\endverse

\beginverse
\[C]Und dann kam der achte, als der Balken krachte,
und sie schrien nach \[G]Klopapier, Klopapier!
\endverse

\beginverse
\[G]Und dann kam der neunte, als die Scheiße schäumte,
und sie schrien nach \[C]Klopapier, Klopapier!
\endverse

\beginverse
\[C]Und dann kam der zehnte, brachte das ersehnte,
Klo – pa – \[G]pier !!
\endverse

\beginverse
\[G]Und dann kam der Elfte, der nahm sich gleich die Hälfte,
und sie schrien nach \[C]Klopapier, Klopapier.
\endverse

\beginverse
\[C]Und dann kam der Zwölfte, der nahm die andre Hälfte,
und sie schrien nach \[G]Klopapier, Klopapier.
\endverse

\beginverse
\[G]Saßen zehn Gestalten, auf dem Donnerbalken,
und sie schrien nach \[C]Klopapier, \[G]KLO-\[G]PA-\[C]PIER!
\endverse
